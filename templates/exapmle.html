import pandas as pd
import numpy as np
from skyulf.automl import AutoFeatureEngineer

# Load your dataset
df = pd.read_csv('your_data.csv')

# Initialize Auto Feature Engineer
afe = AutoFeatureEngineer(llm_assist=True)

# Automated feature engineering
enhanced_df = afe.transform(df)

# View AI-generated feature explanations
print(afe.get_feature_explanations())

# Export to model training
enhanced_df.to_csv('enhanced_features.csv', index=False)
